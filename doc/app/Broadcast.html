def self.today(channel_id=nil)
  if channel_id
  self.where(:channel_id => channel_id).
          where("start >= ? and end <= ?", 
                Time.now - SECONDS_BEFORE_FILTER, 
                Time.now + SECONDS_AFTER_FILTER)
  else
    self.where("start >= ? and end <= ?", 
                Time.now - SECONDS_BEFORE_FILTER, 
                Time.now + SECONDS_AFTER_FILTER)   
  end
end
